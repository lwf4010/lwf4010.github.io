三.mysql的使用

打开mysql

  1. 启动mysql
  2. 打开此电脑，进入D:\phpstudy_pro\Extensions\MySQL5.7.26\bin
  3. 输入cmd

登录mysql

  1. mysql -u root -p
  2. root

mysql基础命令

  1. 查看所有数据库,;是sql语句的结束符号
    show databases;

  2. 创建study数据库
    CREATE DATABASE study;
    出现Query OK, 1 row affected (0.01 sec)表示执行成功

  3. 进入study数据库
    use study;
    出现Database changed表示执行成功

  4. 查看所有数据表
    show tables;

  5. 创建users表

  6. 查看users表中所有的内容
    select * from users;
    select 查询想要的字段
    *所有字段
    from 来自
    users表

在users表中插入数据

INSERT INTO users (username, password) VALUES ('zhangsan', '123456');
插入 表 字段1，字段2 值 （值1，值2;

修改users表中zhangsan的数据，改成admin，admin

UPDATE users SET username = 'admin', password = 'admin' WHERE username = 'zhangsan';

删除users表

DROP TABLE users;

删除数据库

DROP DATABASE study;

报错处理

  1. mysql远程连接
    1. grant all privileges on *.* to 'root'@'%' identified by 'root' with grant option;
    2. flush privileges;
